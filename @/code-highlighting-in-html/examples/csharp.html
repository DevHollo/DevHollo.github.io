using System;

class Program
{
    static void Main(string[] args)
    {
        int number = 5;
        long factorial = CalculateFactorial(number);
        Console.WriteLine($"Factorial of {number} is: {factorial}");
    }

    static long CalculateFactorial(int n)
    {
        if (n == 0)
            return 1;
        else
            return n * CalculateFactorial(n - 1);
    }
}
